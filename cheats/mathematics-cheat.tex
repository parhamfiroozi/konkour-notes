\section{Engineering Mathematics}
\begin{CheatBox}
\textbf{Fourier Series (Period-Shift Integral)}\newline
If $f(t+2p)=f(t)$, then
\[
\int_d^{d+2p} f(t)\,dt = \int_0^{2p} f(t)\,dt.
\]

\textbf{Integration by Parts}\newline
\[
\int u\,dv = uv - \int v\,du.
\]

\textbf{Exponential--Trigonometric Integrals}\newline
For $a,b\in\mathbb{R}$ with $a^2+b^2\neq 0$,
\[
\int e^{ax}\cos(bx)\,dx = \frac{e^{ax}}{a^2+b^2}\bigl(a\cos(bx)+b\sin(bx)\bigr) + C,
\]
\[
\int e^{ax}\sin(bx)\,dx = \frac{e^{ax}}{a^2+b^2}\bigl(a\sin(bx)-b\cos(bx)\bigr) + C.
\]
\end{CheatBox}
